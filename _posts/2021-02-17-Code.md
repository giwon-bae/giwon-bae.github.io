def GuGu(val):
    result = []
    for i in range(1,10):
        result.append(val * i)
    return print(result)

cal1 = GuGu(3)

##3과 5의 배수 합하기##
def Multiply(val):
    result = 0
    for i in range(1,val+1):
        if i % 3 == 0 or i % 5 == 0:
            result += i
        i += 1
    return print(result)

cal2 = Multiply(6)

#-------------------------------------
##게시판 페이징하기##
def Paging(all_num, show_num):
    if all_num % show_num == 0:
        return print(all_num // show_num)
    else:
        return print(all_num // show_num + 1)

cal3 = Paging(29, 5)
